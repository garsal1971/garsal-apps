Salva prompt e risposte di Claude ricevute da Weight Quest.
Un record per ogni chiamata andata a buon fine.
SQL PROD ps_diet_consultations 
CREATE TABLE IF NOT EXISTS ps_diet_consultations (
  id          uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id     uuid        NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE
                          DEFAULT auth.uid(),
  prompt      text        NOT NULL,
  response    text        NOT NULL,
  created_at  timestamptz NOT NULL DEFAULT now()

CREATE INDEX IF NOT EXISTS ps_diet_consultations_user_date_idx

CREATE TABLE cm_profile — profilo personale utente
Dati anagrafici e clinici (altezza, patologie, farmaci). Un record per utente.
SQL PROD cm_profile 
CREATE TABLE IF NOT EXISTS cm_profile (
  id            uuid        PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       uuid        NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE
                            DEFAULT auth.uid(),
  nome          text,
  cognome       text,
  email         text,
  avatar_url    text,
  data_nascita  date,
  sesso         text        CHECK (sesso IN ('M', 'F', 'Altro')),
  altezza_cm    smallint    CHECK (altezza_cm > 50 AND altezza_cm < 300),
  patologie     text,
  farmaci       text,
  note_mediche  text,
  created_at    timestamptz NOT NULL DEFAULT now(),
  updated_at    timestamptz NOT NULL DEFAULT now(),
  UNIQUE (user_id)
);

CREATE OR REPLACE FUNCTION cm_profile_set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END;
$$;

CREATE TRIGGER cm_profile_updated_at
  BEFORE UPDATE ON cm_profile
  FOR EACH ROW EXECUTE FUNCTION cm_profile_set_updated_at();

GRANT ALL ON TABLE cm_profile TO authenticated;
GRANT ALL ON TABLE cm_profile TO service_role;

ALTER TABLE cm_profile ENABLE ROW LEVEL SECURITY;

CREATE POLICY "profilo: select" ON cm_profile FOR SELECT TO authenticated
  USING (auth.uid() = user_id);
CREATE POLICY "profilo: insert" ON cm_profile FOR INSERT TO authenticated
  WITH CHECK (auth.uid() = user_id);
CREATE POLICY "profilo: update" ON cm_profile FOR UPDATE TO authenticated
  USING (auth.uid() = user_id);
CREATE POLICY "profilo: delete" ON cm_profile FOR DELETE TO authenticated
  USING (auth.uid() = user_id);
