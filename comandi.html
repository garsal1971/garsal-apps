Aggiunge 'smart_block' al CHECK di cm_notification_rules.channel, la colonna notification_channel su ts_tasks, e le RLS per il client Android (anon key) sulla queue.

-- 1. cm_notification_rules.channel: aggiunge 'smart_block'
ALTER TABLE cm_notification_rules
    DROP CONSTRAINT IF EXISTS cm_notification_rules_channel_check;

ALTER TABLE cm_notification_rules
    ADD CONSTRAINT cm_notification_rules_channel_check
        CHECK (channel IN ('telegram', 'browser', 'push', 'smart_block'));

-- 2. ts_tasks: colonna notification_channel
ALTER TABLE ts_tasks
    ADD COLUMN IF NOT EXISTS notification_channel text
        NOT NULL DEFAULT 'telegram'
        CHECK (notification_channel IN ('telegram', 'smart_block'));

-- 3. cm_notification_queue: RLS per client Android (anon key)
--    DROP IF EXISTS per idempotenza (eseguibile più volte senza errori)
DROP POLICY IF EXISTS "smart_block_anon_select" ON cm_notification_queue;
DROP POLICY IF EXISTS "smart_block_anon_update" ON cm_notification_queue;

CREATE POLICY "smart_block_anon_select"
    ON cm_notification_queue
    FOR SELECT
    TO anon
    USING (channel = 'smart_block');

CREATE POLICY "smart_block_anon_update"
    ON cm_notification_queue
    FOR UPDATE
    TO anon
    USING  (channel = 'smart_block')
    WITH CHECK (
        channel = 'smart_block'
        AND status IN ('sent', 'failed')
    );

🔍 Verifica RLS Smart Block

Esegui in Supabase SQL Editor per verificare lo stato delle policy:

-- Controlla se le policy esistono
SELECT policyname, roles, cmd, qual
FROM pg_policies
WHERE tablename = 'cm_notification_queue'
ORDER BY policyname;

-- Controlla se RLS è abilitato sulla tabella
SELECT relname, relrowsecurity AS rls_enabled
FROM pg_class
WHERE relname = 'cm_notification_queue';

-- Conta righe smart_block nella queue per stato
SELECT status, count(*)
FROM cm_notification_queue
WHERE channel = 'smart_block'
GROUP BY status;
