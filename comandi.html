FIX — DecisionKeeper: user_id + RLS su dc_decisions e dc_logs
Aggiunge user_id a dc_logs, abilita RLS su entrambe le tabelle e crea le policy per isolare i dati per utente.
FIX SQL PROD dc_decisions · dc_logs 
-- 1. Rendi user_id NOT NULL con default auth.uid() su dc_decisions
ALTER TABLE dc_decisions
  ALTER COLUMN user_id SET DEFAULT auth.uid();

-- 2. Abilita RLS
ALTER TABLE dc_decisions ENABLE ROW LEVEL SECURITY;
ALTER TABLE dc_logs      ENABLE ROW LEVEL SECURITY;

-- 3. Policy dc_decisions
CREATE POLICY "decisions: select" ON dc_decisions FOR SELECT TO authenticated
  USING (auth.uid() = user_id);
CREATE POLICY "decisions: insert" ON dc_decisions FOR INSERT TO authenticated
  WITH CHECK (auth.uid() = user_id);
CREATE POLICY "decisions: update" ON dc_decisions FOR UPDATE TO authenticated
  USING (auth.uid() = user_id);
CREATE POLICY "decisions: delete" ON dc_decisions FOR DELETE TO authenticated
  USING (auth.uid() = user_id);

-- 4. Policy dc_logs — usa la relazione con dc_decisions
CREATE POLICY "logs: select" ON dc_logs FOR SELECT TO authenticated
  USING (EXISTS (
    SELECT 1 FROM dc_decisions
    WHERE dc_decisions.id = dc_logs.decision_id
      AND dc_decisions.user_id = auth.uid()
  ));
CREATE POLICY "logs: insert" ON dc_logs FOR INSERT TO authenticated
  WITH CHECK (EXISTS (
    SELECT 1 FROM dc_decisions
    WHERE dc_decisions.id = dc_logs.decision_id
      AND dc_decisions.user_id = auth.uid()
  ));
CREATE POLICY "logs: delete" ON dc_logs FOR DELETE TO authenticated
  USING (EXISTS (
    SELECT 1 FROM dc_decisions
    WHERE dc_decisions.id = dc_logs.decision_id
      AND dc_decisions.user_id = auth.uid()
  ));

FIX — Ricrea cron fill-notification-queue (Vault auth)
Il cron fallisce con unrecognized configuration parameter "app.service_role_key" perché usa current_setting() invece del Vault. Questo comando reschedula il job leggendo la chiave da vault.decrypted_secrets.
FIX SQL PROD cron.job 
SELECT cron.unschedule('fill-notification-queue');

SELECT cron.schedule(
    'fill-notification-queue',
    '0 */6 * * *',
    $$
    SELECT net.http_post(
        url     := 'https://jajlmmdsjlvzgcxiiypk.supabase.co/functions/v1/fill-notification-queue',
        headers := jsonb_build_object(
            'Content-Type',  'application/json',
            'Authorization', 'Bearer ' || (
                SELECT decrypted_secret
                FROM   vault.decrypted_secrets
                WHERE  name = 'supabase_service_role_key'
                LIMIT  1
            )
        ),
        body    := '{}'::jsonb
    ) AS request_id;
    $$
);
